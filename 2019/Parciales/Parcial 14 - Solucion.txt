ORG 3500H
MULT1: PUSH BX
PUSH CX
MOV BX,AX
MOV CX, 0
SIGUE: CMP BX, 0
JZ FFIN
ADD CX, DX
dec BX
jmp SIGUE
FFIN: MOV DX,CX
POP CX
POP BX
RET

ORG 3000H
POT: PUSH BX
MOV BX, SP
ADD BX, 4
PUSH CX
PUSH DX

MOV CX,[BX] ; RECIBIR PARÁMETROS
MOV DX, 1
ITER: CMP CX, 0
JZ FINR
CALL MULT1
DEC CX
JMP ITER
FINR: MOV [BX],DX
POP DX
POP CX
POP BX
RET
